// ✅ ОБРАБОТКА ПАПОК — исправленная версия async function processFolder(rootPublicKey, targetPath, relativePath = '') { if (!isRunning) throw new Error('Остановлено'); const folderKey = extractPublicKey(rootPublicKey); const fullRelativePath = relativePath ? `${relativePath}/` : ''; addLog(`📂 Обработка: ${targetPath} (ключ: ${folderKey})`); updateStatus(`Обработка: ${targetPath}`); await ensureFolder(targetPath); let url = `${API}/public/resources?public_key=${encodeURIComponent(folderKey)}&limit=1000`; if (relativePath) url += `&path=${encodeURIComponent(relativePath)}`; let hasMore = true; while (hasMore && isRunning) { const resp = await fetch(url, { signal: abortController?.signal }); if (!resp.ok) throw new Error(`Не получено содержимое (${resp.status})`); const data = await resp.json(); const items = data._embedded?.items || []; addLog(`📊 Найдено: ${items.filter(i=>i.type==='file').length} файлов, ${items.filter(i=>i.type==='dir').length} папок`); for (const item of items) { if (!isRunning) throw new Error('Остановлено'); if (item.type === 'file') { // ✅ Используем rootPublicKey + относительный путь файла const fileRelativePath = `${fullRelativePath}${item.name}`; await savePublicFile(rootPublicKey, targetPath, item.name, fileRelativePath); } else if (item.type === 'dir') { // ✅ Рекурсивно обрабатываем подпапку const newRelativePath = `${fullRelativePath}${item.name}`; const newTargetPath = `${targetPath}/${item.name}`; await ensureFolder(newTargetPath); await processFolder(rootPublicKey, newTargetPath, newRelativePath); } } hasMore = !!(data._embedded?.next || data.next); url = data._embedded?.next || data.next || ''; if (hasMore && !url) break; } }
